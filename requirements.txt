# =============================================
# 24/7 YouTube Live Stream - Requirements
# =============================================
# These are system-level dependencies installed
# via apt-get in the GitHub Actions workflow.
# No Python packages are required.

# --- System Dependencies (Ubuntu/Debian) ---
# ffmpeg          >= 4.4      (video encoding & streaming)
# jq              >= 1.6      (JSON parsing for playlist.json)
# wget            >= 1.21     (logo image download if URL given)
# fonts-dejavu-core           (fonts for ffmpeg drawtext ticker)

# --- GitHub Actions ---
# actions/checkout@v4         (repo checkout)
# ubuntu-latest runner        (build environment)

# --- Required GitHub Secrets ---
# YOUTUBE_STREAM_KEY          (YouTube Live Stream Key from YouTube Studio)

# --- Required Repository Files ---
# stream.sh                   (main streaming script)
# playlist.json               (video URLs, logo, ticker, effects config)
# logo.png                    (logo image — OR set logo.path to a URL in playlist.json)

# --- YouTube Live Settings (recommended) ---
# Resolution : 1280x720 or 1920x1080
# Frame Rate  : 30fps
# Video Bitrate: 4500 kbps
# Audio Bitrate: 192 kbps
# Encoder     : libx264
# Stream URL  : rtmp://a.rtmp.youtube.com/live2/<STREAM_KEY>
